----------------------Export-data-from-sql-server-to-parquet
-- 1. Install and load the ODBC extension
INSTALL odbc;
LOAD odbc;

-- 2. Export the SQL Server table directly to a Parquet file
COPY (
    SELECT * FROM odbc_query(
        'Driver={ODBC Driver 18 for SQL Server};Server=YOUR_SERVER_NAME;Database=YOUR_DB_NAME;Trusted_Connection=yes;TrustServerCertificate=yes;', 
        'SELECT * FROM your_table_name'
    )
) TO 'output_file.parquet' (FORMAT 'PARQUET', COMPRESSION 'ZSTD');

-- Read and view the entire Parquet file
SELECT * FROM read_parquet('output_file.parquet');

-- Alternative shorter syntax (DuckDB auto-detects the file type)
SELECT * FROM 'output_file.parquet';
